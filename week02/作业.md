# UTF-8 Encoding
    function fn(str) {
        str = str.trim();
        let arr = [];
        for (let i = 0; i < str.length; i++) {
            if (str[i])
                arr.push('\\u' + str[i].charCodeAt().toString(16))
        }
        return arr.join('');
    }
